use zk_identity_masks::{Mask, RegistryClient};

#[tokio::main]
async fn main() {
    let mut age = [0u8; 32];
    age[0] = 25;
    let mut country = [0u8; 32];
    country[0] = 840;

    let mask = Mask::new(vec![age, country], None);
    println!("Commitment: {}", hex::encode(mask.commitment()));

    // Generate proof
    let proof = mask.create_proof("age >= 18", &[0u8; 32]).await;
    println!("Proof: {:?}", proof);

    // Verify using Zig verifier
    let valid = proof.verify().await;
    println!("Valid: {}", valid);
}
